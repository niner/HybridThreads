from threading import Thread
from time import sleep

starter = 0
results = []
threads = []

def sayer(name, results):
    while starter == 0:
        sleep(0.1)
    results.append(name)

for i in range(10000):
    t = Thread(target = sayer, args = (i, results))
    t.start();
    threads.append(t)

starter = 1

for t in threads:
    t.join()

print len(results)
